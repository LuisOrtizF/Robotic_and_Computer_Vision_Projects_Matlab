 2function E = chessboardEnergy(chessboard,corners)

 4    % energy: number of corners
 5    E_corners = -size(chessboard,1)*size(chessboard,2);
 6

 8    E_structure = 0;

12      for k=1:size(chessboard,2)-2

15      end
16    end
17
18    % walk through columns
19    for j=1:size(chessboard,2)

22        E_structure = max(E_structure,norm(x(1,:)+x(3,:)-2*x(2,:))/norm(x(1,:)-x(3,:)));
23      end
24    end
25
26    % final energy
27    E = E_corners + 1*size(chessboard,1)*size(chessboard,2)*E_structure;
