43function [neighbor_idx,min_dist] = directionalNeighbor(idx,v,chessboard,corners)
44
45    % list of neighboring elements, which are currently not in use
46    unused       = 1:size(corners.p,1);
47    used         = chessboard(chessboard~=0);
48    unused(used) = [];
49    coder.varsize('unused');
50
51    % direction and distance to unused corners
52    dir  = corners.p(unused,:) - ones(length(unused),1)*corners.p(idx,:);
53    dist = (dir(:,1)*v(1)+dir(:,2)*v(2));
54
55    % distances
56    dist_edge = dir-dist*v;
57    dist_edge = sqrt(dist_edge(:,1).^2+dist_edge(:,2).^2);
58    dist_point = dist;
59    dist_point(dist_point<0) = inf;
60
61    % find best neighbor
62    [min_dist,min_idx] = min(dist_point+5*dist_edge);
63    neighbor_idx = unused(min_idx);
64end
