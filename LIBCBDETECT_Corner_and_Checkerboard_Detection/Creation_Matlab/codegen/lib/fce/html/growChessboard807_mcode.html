  2function chessboard = growChessboard(chessboard,corners,border_type)
  3
  4    % return immediately, if there do not exist any chessboards
  5    if isempty(chessboard)
  6      return;
  7    end
  8
  9    % extract feature locations
 10    p = corners.p;
 11
 12    % list of unused feature elements
 13    unused       = 1:size(corners.p,1);
 14    used         = chessboard(chessboard~=0);
 15    unused(used) = [];
 16    coder.varsize('unused');
 17
 18    % candidates from unused corners
 19    cand = p(unused,:);
 20
 21    % switch border type 1..4
 22    switch border_type
 23
 24      case 1
 25        pred = predictCorners(p(chessboard(:,end-2),:),p(chessboard(:,end-1),:),p(chessboard(:,end),:));
 26        idx = assignClosestCorners(cand,pred);
 27        if all(idx~=0)
 28          chessboard = [chessboard unused(idx)'];
 29        end
 30
 31      case 2
 32        pred = predictCorners(p(chessboard(end-2,:),:),p(chessboard(end-1,:),:),p(chessboard(end,:),:));
 33        idx = assignClosestCorners(cand,pred);
 34        if all(idx~=0)
 35          chessboard = [chessboard; unused(idx)];
 36        end
 37
 38      case 3
 39        pred = predictCorners(p(chessboard(:,3),:),p(chessboard(:,2),:),p(chessboard(:,1),:));
 40        idx = assignClosestCorners(cand,pred);
 41        if all(idx~=0)
 42          chessboard = [unused(idx)' chessboard];
 43        end
 44
 45      case 4
 46        pred = predictCorners(p(chessboard(3,:),:),p(chessboard(2,:),:),p(chessboard(1,:),:));
 47        idx = assignClosestCorners(cand,pred);
 48        if all(idx~=0)
 49          chessboard = [unused(idx); chessboard];
 50        end
 51    end
 52end
