 2function chessboards = chessboardsFromCorners(corners)
 3
 4    % intialize chessboards
 5    n=5;
 6    assert(n<6)
 7    chessboards = cell(1,n);
 8
 9    for i = 1:n
10        chessboards{i} = zeros(6,7);
11    end
12     
13    p_energy=zeros(1,4);
14    proposal = {0 0 0 0};
15    coder.varsize('proposal');
16
17    a=0;
18    
19    % for all seed corners do
20    for i=1:size(corners.p,1)
21
22        % init 3x3 chessboard from seed i
23      chessboard = initChessboard(corners,i);
24      coder.varsize('chessboard'); 
25
26      % check if this is a useful initial guess
27      if isempty(chessboard) || chessboardEnergy(chessboard,corners)>0
28        continue;
29      end
30
31      % try growing chessboard
32      while 1
33
34        % compute current energy
35        energy = chessboardEnergy(chessboard,corners);
36
37        % compute proposals and energies
38        for j=1:4
39          proposal{j} = growChessboard(chessboard,corners,j);
40          p_energy(j) = chessboardEnergy(proposal{j},corners);
41        end
42
43        % find best proposal
44        [~,min_idx] = min(p_energy);
45
46        % accept best proposal, if energy is reduced
47        if p_energy(min_idx)<energy
48          chessboard = proposal{min_idx};
49
50        % otherwise exit loop
51        else
52          break;
53        end
54      end
55      
56      % if chessboard has low energy (corresponding to high quality)
57      if chessboardEnergy(chessboard,corners)<-10
58
59        if a==0
60            chessboards{1} = zeros(size(chessboard));
61        end
62
63        % check if new chessboard proposal overlaps with existing chessboards
64        overlap = zeros(length(chessboards),2);
65        
66        %[rr,cc]=size(chessboards);
67        
68        for j=1:length(chessboards)     %rr*cc     
69          for k=1:length(chessboards{j}(:))
70            if any(chessboards{j}(k)==chessboard(:))
71              overlap(j,1) = 1;
72              overlap(j,2) = chessboardEnergy(chessboards{j},corners);
73              break;
74            end
75          end
76        end
77
78        % add chessboard (and replace overlapping if neccessary)
79        if ~any(overlap(:,1))
80          a=a+1;
81          chessboards{a} = chessboard;
82        else
83          idx = find(overlap(:,1)==1);
84          if ~any(overlap(idx,2)<=chessboardEnergy(chessboard,corners))
85            chessboards{idx} = [];
86            chessboards{a} = chessboard;
87          end
88        end
89      end
90
91    end
92    
93    %chessboards = cellfun(@(x) x(any(x,2),:),chessboards,'UniformOutput',false);
94    %chessboards(:, any(cellfun(@isempty, chessboards), 1)) = [];
95    %disp('Find Chessboards OK');  
96end
