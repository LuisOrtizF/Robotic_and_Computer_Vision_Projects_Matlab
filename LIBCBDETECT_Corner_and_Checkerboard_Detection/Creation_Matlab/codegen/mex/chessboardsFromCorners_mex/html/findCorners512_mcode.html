  2function corners3 = findCorners(img, tau, refine_corners)
  3
  4    img = im2double(img);
  5
  6    % 3 scales
  7    radius=zeros(1,3);
  8    radius(1) = 4;
  9    radius(2) = 8;
 10    radius(3) = 12;
 11
 12    % sobel masks
 13    du = [-1 0 1; -1 0 1; -1 0 1];
 14    dv = du';
 15
 16    % compute image derivatives (for principal axes estimation)
 17    img_du     = conv2(double(img),du,'same');
 18    img_dv     = conv2(double(img),dv,'same');
 19       
 20    img_angle  = atan2(img_dv,img_du);
 21    img_weight = sqrt(img_du.^2+img_dv.^2);
 22
 23    % correct angle to lie in between [0,pi]
 24    img_angle(img_angle<0)  = img_angle(img_angle<0)+pi;
 25    img_angle(img_angle>pi) = img_angle(img_angle>pi)-pi;
 26
 27    % scale input image
 28    img     = double(img);
 29    img_min = min(img(:));
 30    img_max = max(img(:));
 31    img     = (img-img_min)/(img_max-img_min);
 32
 33    % template properties
 34    template_props = [0 pi/2 radius(1); pi/4 -pi/4 radius(1); 0 pi/2 radius(2); pi/4 -pi/4 radius(2); 0 pi/2 radius(3); pi/4 -pi/4 radius(3)];
 35
 36    %disp('Filtering ...');
 37
 38    % filter image
 39    img_corners = zeros(size(img,1),size(img,2));
 40
 41    for template_class=1:size(template_props,1)
 42
 43      % create correlation template
 44      template = createCorrelationPatch(template_props(template_class,1),template_props(template_class,2),template_props(template_class,3));
 45
 46      % filter image according with current template
 47      img_corners_a1 = conv2(img,template.a1,'same');
 48      img_corners_a2 = conv2(img,template.a2,'same');
 49      img_corners_b1 = conv2(img,template.b1,'same');
 50      img_corners_b2 = conv2(img,template.b2,'same');
 51
 52      % compute mean
 53      img_corners_mu = (img_corners_a1+img_corners_a2+img_corners_b1+img_corners_b2)/4;
 54
 55      % case 1: a=white, b=black
 56      img_corners_a = min(img_corners_a1-img_corners_mu,img_corners_a2-img_corners_mu);
 57      img_corners_b = min(img_corners_mu-img_corners_b1,img_corners_mu-img_corners_b2);
 58      img_corners_1 = min(img_corners_a,img_corners_b);
 59
 60      % case 2: b=white, a=black
 61      img_corners_a = min(img_corners_mu-img_corners_a1,img_corners_mu-img_corners_a2);
 62      img_corners_b = min(img_corners_b1-img_corners_mu,img_corners_b2-img_corners_mu);
 63      img_corners_2 = min(img_corners_a,img_corners_b);
 64
 65      % update corner map
 66      img_corners = max(img_corners,img_corners_1);
 67      img_corners = max(img_corners,img_corners_2);
 68    end
 69
 70    % extract corner candidates via non maximum suppression
 71    corners.p = nonMaximumSuppression(img_corners,3,0.025,5);
 72       
 73    %disp('Refining ...');
 74
 75    corners2 = struct('p',zeros(size(corners.p,1),2),'v1',zeros(size(corners.p,1),2),'v2',zeros(size(corners.p,1),2));
 76    coder.varsize('corners2.p');
 77    coder.varsize('corners2.v1');
 78    coder.varsize('corners2.v2');
 79    
 80    corners2.p=corners.p;
 81    
 82    % subpixel refinement
 83    if refine_corners
 84      corners2 = refineCorners(img_du,img_dv,img_angle,img_weight,corners2,10);
 85    end
 86       
 87    % remove corners without edges
 88    idx = (corners2.v1(:,1)==0 & corners2.v1(:,2)==0);
 89    
 90    corners2.p(idx,:)  = [];
 91    corners2.v1(idx,:) = [];
 92    corners2.v2(idx,:) = [];
 93
 94    %disp('Scoring ...');
 95   
 96    corners3 = struct('p',zeros(size(corners2.p,1),2),'v1',zeros(size(corners2.v1,1),2),'v2',zeros(size(corners2.v2,1),2),'score',zeros(size(corners2.p,1),1));
 97    coder.varsize('corners3.p');
 98    coder.varsize('corners3.v1');
 99    coder.varsize('corners3.v2');
100    coder.varsize('corners3.score')
101    
102    corners3.p=corners2.p;
103    corners3.v1=corners2.v1;
104    corners3.v2=corners2.v2;
105    
106    % score corners
107    corners3 = scoreCorners(img,img_angle,img_weight,corners3,radius);
108     
109    % remove low scoring corners
110    idx = corners3.score<tau;
111
112    corners3.p(idx,:)     = [];
113    corners3.v1(idx,:)    = [];
114    corners3.v2(idx,:)    = [];
115    corners3.score(idx) = [];
116        
117    % make v1(:,1)+v1(:,2) positive (=> comparable to c++ code)
118    idx = corners3.v1(:,1)+corners3.v1(:,2)<0;
119    corners3.v1(idx,:) = -corners3.v1(idx,:);
120
121    % make all coordinate systems right-handed (reduces matching ambiguities from 8 to 4)
122    corners_n1 = [corners3.v1(:,2) -corners3.v1(:,1)];
123    flipx       = -sign(corners_n1(:,1).*corners3.v2(:,1)+corners_n1(:,2).*corners3.v2(:,2));
124    corners3.v2 = corners3.v2.*(flipx*ones(1,2));
125
126    % convert to 0-based index
127    corners3.p = corners3.p-1;  
128        
129    %disp('Find Corners OK');
130end
