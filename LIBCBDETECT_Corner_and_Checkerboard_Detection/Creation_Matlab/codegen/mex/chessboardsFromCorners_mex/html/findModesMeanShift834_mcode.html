 2function [modes2, hist_smoothed] = findModesMeanShift(hist,sigma)
 3    
 4    modes2=[];
 5    coder.varsize('modes2');
 6
 7    % efficient mean-shift approximation by histogram smoothing
 8    % compute smoothed histogram
 9    hist_smoothed = zeros(1,length(hist));
10    
11    for i=1:length(hist)
12      j                = -round(2*sigma):+round(2*sigma);
13      idx              = mod(i+j-1,length(hist))+1;
14      hist_smoothed(i) = sum(hist(idx).*normpdf(j,0,sigma));
15    end
16
17    % check if at least one entry is non-zero
18    % (otherwise mode finding may run infinitly)
19    if all(abs(hist_smoothed-hist_smoothed(1))<1e-5)
20      return;
21    end
22
23    % mode finding
24    for i=1:length(hist_smoothed)
25      j=i;
26      while 1
27        h0 = hist_smoothed(j);
28        j1 = mod(j+1-1,length(hist))+1;
29        j2 = mod(j-1-1,length(hist))+1;
30        h1 = hist_smoothed(j1);
31        h2 = hist_smoothed(j2);
32        if h1>=h0 && h1>=h2
33          j=j1;
34        elseif h2>h0 && h2>h1
35          j=j2;
36        else
37          break;
38        end
39      end
40      if size(modes2,1)==0 ||~any(modes2(:,1)==j)
41        modes2 = [modes2; j hist_smoothed(j)];
42      end
43    end
44
45    % sort
46    [~, idx] = sort(modes2(:,2),1,'descend');
47    modes2 = modes2(idx,:);
48end
