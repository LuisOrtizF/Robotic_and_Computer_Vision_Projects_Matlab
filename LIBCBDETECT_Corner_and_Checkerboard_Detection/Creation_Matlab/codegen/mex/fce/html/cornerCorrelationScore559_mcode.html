1
 2function score = cornerCorrelationScore(img,img_weight,v1,v2)
 3
 4    % center
 5    c = ones(1,2)*(size(img_weight,1)+1)/2;
 6
 7    % compute gradient filter kernel (bandwith = 3 px)
 8    img_filter = -1*ones(size(img_weight,1),size(img_weight,2));
 9    for x=1:size(img_weight,2)
10      for y=1:size(img_weight,1)
11        p1 = [x y]-c;
12        p2 = p1*(v1)'*v1;
13        p3 = p1*(v2)'*v2;
14        if norm(p1-p2)<=1.5 || norm(p1-p3)<=1.5
15          img_filter(y,x) = +1;
16        end
17      end
18    end
19
20    % convert into vectors
21    vec_weight = img_weight(:);
22    vec_filter = img_filter(:);
23
24    % normalize
25    vec_weight = (vec_weight-mean(vec_weight))/std(vec_weight);
26    vec_filter = (vec_filter-mean(vec_filter))/std(vec_filter);
27
28    % compute gradient score
29    score_gradient = max(sum(vec_weight.*vec_filter)/(length(vec_weight)-1),0);
30
31    % create intensity filter kernel
32    template = createCorrelationPatch(atan2(v1(2),v1(1)),atan2(v2(2),v2(1)),c(1)-1);
33
34    % checkerboard responses
35    a1 = sum(template.a1(:).*img(:));
36    a2 = sum(template.a2(:).*img(:));
37    b1 = sum(template.b1(:).*img(:));
38    b2 = sum(template.b2(:).*img(:));
39
40    % mean
41    mu = (a1+a2+b1+b2)/4;
42
43    % case 1: a=white, b=black
44    score_a = min(a1-mu,a2-mu);
45    score_b = min(mu-b1,mu-b2);
46    score_1 = min(score_a,score_b);
47
48    % case 2: b=white, a=black
49    score_a = min(mu-a1,mu-a2);
50    score_b = min(b1-mu,b2-mu);
51    score_2 = min(score_a,score_b);
52
53    % intensity score: max. of the 2 cases
54    score_intensity = max(max(score_1,score_2),0);
55
56    % final score: product of gradient and intensity score
57    score = score_gradient*score_intensity;
58end
